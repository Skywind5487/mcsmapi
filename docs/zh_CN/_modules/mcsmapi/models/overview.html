class SystemUser(BaseModel):

class SystemInfo(BaseModel):
    user: SystemUser = SystemUser()

    totalmem: int = 0
    freemem: int = 0
    type: str = ""
    version: str = ""
    node: str = ""
    hostname: str = ""
    loadavg: List[float] = []
    platform: str = ""
    release: str = ""
    uptime: int = 0
    cpu: float = 0
